\section{The Black Scholes Equation and Options}
An option is a financial contract whose value is derived from an underlying asset, such as a stock or a bond. 
An options contract grants the holder the right, but not the obligation, to buy or sell an asset at a predetermined price, 
known as the strike, and at an agreed-upon future date, known as the maturity date. 
\subsection{Derivation of the Black Scholes equation}
Derivation from the GBM
\begin{equation}
    dS(t) = \mu S(t) dt + \sigma S(t) dX(t)
\end{equation}


\begin{equation}
\frac{\partial V}{\partial t} + \frac{1}{2}\sigma^{2} S^{2} \frac{\partial^{2} V}{\partial S^{2}} + r S \frac{\partial V}{\partial S} - r V = 0
     \label{eq:BSE}
\end{equation}
with the boundary conditions:

\begin{equation}
    C\left(0,t\right) = 0 \forall t
\end{equation}

\begin{equation}
    C\left(S,t\right) \approxS-Ke^{-rT-t} \text{ as } S\rightarrow \infty 
\end{equation}


\begin{equation}
    C\left(S,T\right) = \max\left\{S-K,0\right\}
\end{equation}

Alternatively for a put denoted as P(S,t) the boundary conditions are:
\begin{equation}
    P\left(S,t\right) = 0 \forall t \text{ as } S\rightarrow \infty 
\end{equation}

\begin{equation}
    C\left(0,t\right) \approxS Ee^{-r\left( T-t \right)} \forall t
\end{equation}


\begin{equation}
    C\left(S,T\right) = \max\left\{S-K,0\right\}
\end{equation}

For non-dividend-paying European Call Options, the closed form solution to \ref{eq:BSE} is:
\begin{equation}
  C(S,t) = SN(d_1) - Xe^{-r(T-t)}N(d_2)
\end{equation}
and for a put it is:
\begin{equation}
    P(S,t) = Xe^{-r(T-t)}N(-d_2) - SN(-d_1)
\end{equation}
where:

\begin{equation}
    d_1 = \frac{\log(S/X) + (r+ 0.5 \sigma^2)(T-t)} {\sigma \sqrt{T-t}}
\end{equation}

\begin{equation}
    d_1 = \frac{\log(S/X) + (r - 0.5 \sigma^2)(T-t)}{\sigma \sqrt{T-t}}
\end{equation}



\section{The greeks}
To quantify different aspects of risk in an option position, the term the Greeks is used. 
\subsection{Delta}
The delta ($\Delta$) of an options contract is defined as the rate of change of the option price with respect to the price of the underlying asset \cite{alma991008589012108046}. Delta can be defined as:
\begin{equation}
   \Delta =  \frac{\partial V}{\partial S}
\end{equation}
\begin{figure}[htbp]
    \centering
    \input{tikz/Delta.tex}
    \caption{The impact of stock price on the Delta of European Call and Put Option.
}
    \label{fig:delta_greek}
\end{figure}
For a European call, the delta ranges from 0 to 1 with respect to the underlying price, while for a put, it ranges from -1 to 0. As seen in\ref{fig:delta_greek}, delta changes most rapidly near the strike, where the option is at the money.

\subsection{Theta}
The theta ($\Theta$) of an option measures the sensitivity of the option to the passage of time, often referred to as \textit{time decay}. It is defined as:
\begin{equation}
    \Theta = - \frac{\partial V}{\partial t}
     \label{eq:theta}
\end{equation}
The negative sign in \ref{eq:theta} indicates that, all else equal, an option loses value as it approaches maturity. This arises from reduced uncertainty and, therefore, a diminishing value of optionality. As seen in \ref{fig:theta_greeks}, time decay is not linear, and it accelerates as time to expiration approaches. 
\begin{figure}[htbp]
    \centering
    \input{tikz/Theta.tex}
    \caption{The Theta of European Options. }
    \label{fig:theta_greeks}
\end{figure}

\subsection{Gamma}
The gamma $\Gamma$ of an option measures the rate of change in delta with respect to the price of the underlying. It is the second-order derivative with respect to the asset price. 
\begin{equation}
    \Gamma =\frac{\partial^2 V}{\partial S^2}
\end{equation}
Gamma is particularly important for delta hedging, as it quantifies how often a portfolio needs to be rebalanced to remain delta-neutral. A large Gamma indicates that very frequent rebalancing is needed, while a low Gamma indicates the opposite. 
\begin{figure}[htbp]
    \centering
    \input{tikz/Gamma.tex}
    \caption{Gamma of European options.}
    \label{fig:gamma_greek}
\end{figure}
Figure \ref{fig:gamma_greek} shows that Gamma peaks around the Strike price where the option transitions between in the money and out of the money. As it moves further away from the strike, gamma approaches zero, indicating that delta is constant. 

These sensitivities appear directly in the Black-Scholes equation and the BSE can be written as:
\begin{equation}
    \Theta
+ \frac{1}{2}\sigma^2 S^2 \Gamma
+ rS\Delta
- rV = 0,
\end{equation}

highlighting the fundamental role of the Greeks in option pricing dynamics.

\section{Finite Difference method for the BSE}

\subsection{Explicit}

\begin{equation}
    \frac{\partial V}{\partial t}
+ \frac{1}{2}\sigma^{2} S^{2} \frac{\partial^{2} V}{\partial S^{2}}
+ r S \frac{\partial V}{\partial S}
- r V = 0
\end{equation}

$\frac{\partial V}{\partial t}$ rate of change of option value with respect to time: 
\begin{equation}
\frac{V_{i,k} - V_{i,k+1}}{\Delta t}
\end{equation}

$\frac{\partial^{2} V}{\partial S^{2}}$ second derivative with respect to stock price:
\begin{equation}
\frac{V_i+1,_k - 2V_i,_k  + V_i-1_,k}{\Delta S^2}
\end{equation}



$\frac{\partial V}{\partial S}$ Delta first derivative with respect to stock price:
\begin{equation}
\frac{V_{i+1,k} - V_{i-1,k}}{2\Delta S}
\end{equation}

- Starting with the BSE, substitute in the derivatives

\begin{equation}
  \frac{V_{i,k} - V_{i,k+1}}{\Delta t} + \frac{1}{2} \sigma^2 S_i^2 \frac{V_i+1,_k - 2V_i,_k  + V_i-1_,k}{\Delta S^2} + rS_i \frac{V_{i+1,k} - V_{i-1,k}}{2\Delta S} - rV_{i,k} = 0
\end{equation}

Move the first derivative to the lhs to solve for $V_{i,k}$
\begin{equation}
V_{i,k}  = V_{i,k+1} +  \Delta t + \frac{1}{2} \sigma^2 S_i^2 \frac{V_{i+1,k+1} - 2V{_i,_{k+1}}  + V_{i-1_,k+1}}{\Delta S^2} + rS_i \frac{V_{i+1,k+1} - V_{i-1,k+1}}{2\Delta S} - rV_{i,k+1}
\end{equation}

Collect the coefficients:
\begin{equation}
  V_{s,t} = A_t V_{s+1,t-1} + B_t V_{s+1, t} + C_t V_{s+1,t+1}
\end{equation}


where:
\begin{equation}
A_t = \frac{1}{2} \Delta t \left(\sigma^2 i^2 - r i\right)
\end{equation}
\begin{equation}
B_t = 1 - \Delta t \left(\sigma^2 i^2 + r\right)
\end{equation}
\begin{equation}
C_t = \frac{1}{2} \Delta t \left(\sigma^2 i^2 + r i\right)
\end{equation}




The option price today is $V_{s, 0}$ where $s$ corresponds to the current stock price
